add spork to Gemfile
gem 'spork',        '0.9.0.rc9'

sudo bundle install

add following line to .rspec
--drb

bootstrap spec/spec_helper.rb file
bundle exec spork -b

edit spec/spec_helper.rb as needed

run spork
bundle exec spork

run rspec
rspec spec/
